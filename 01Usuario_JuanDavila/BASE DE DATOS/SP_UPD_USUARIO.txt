ALTER PROCEDURE [dbo].[SP_UPD_USUARIO]
    @PI_ID_USUARIO INTEGER,
    @PV_CODIGO VARCHAR (20),
	@PV_NOMBRES VARCHAR (100),
	@PV_APELLIDOS VARCHAR (100),
	@PV_MAIL VARCHAR (100),
	@PD_FECHA_NACIMIENTO DATETIME,
	@PV_CONTRASENA VARCHAR (100),
	@PI_USUARIO_MODIFICACION INTEGER


	
AS
DECLARE @CONTADOR_ID INTEGER
BEGIN
     UPDATE USUARIO
	 SET CODIGO=@PV_CODIGO,
	     NOMBRES=@PV_NOMBRES,
	     APELLIDOS=@PV_APELLIDOS,
	     MAIL=@PV_MAIL,
	     FECHA_NACIMIENTO=@PD_FECHA_NACIMIENTO,
	     CONTRASENA=@PV_CONTRASENA,
	     USUARIO_MODIFICACION=@PI_USUARIO_MODIFICACION,
	     FECHA_MODIFICACION=GETDATE()
     WHERE
	 ID_USUARIO=@PI_ID_USUARIO
END
